部署网络命令速查

这些命令不是让你照抄上线，而是帮你快速定位“流量到哪一层断了”。生产环境使用前先确认绑定地址、认证和防火墙。

反向代理与 TLS

# Caddy：公网 HTTPS → 本机应用
caddy reverse-proxy --from app.example.com --to 127.0.0.1:3000

# Caddyfile
app.example.com {
  reverse_proxy 127.0.0.1:3000
}

# Nginx：最小反向代理骨架
server {
  listen 443 ssl http2;
  server_name app.example.com;

  ssl_certificate     /etc/letsencrypt/live/app.example.com/fullchain.pem;
  ssl_certificate_key /etc/letsencrypt/live/app.example.com/privkey.pem;

  location / {
    proxy_pass http://127.0.0.1:3000;
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;
  }
}

正向代理与 SOCKS5

# curl 走 HTTP 正向代理
curl -v -x http://proxy.example.com:3128 https://example.com/

# curl 走 SOCKS5；socks5h 让代理端解析 DNS
curl -v --proxy socks5h://127.0.0.1:1080 https://example.com/

# SSH 动态转发：本机开 SOCKS5 代理
ssh -D 127.0.0.1:1080 user@jump-host

SSH Tunnel

# 本地转发：本机 15432 → 远端内网 db.internal:5432
ssh -L 127.0.0.1:15432:db.internal:5432 user@jump-host

# 远程转发：公网服务器 18080 → 内网本机 8080
ssh -R 127.0.0.1:18080:127.0.0.1:8080 user@public-server

# 跳板机：经 bastion 登录 target
ssh -J user@bastion user@target.internal

# 只做转发，不开远端 shell
ssh -N -L 127.0.0.1:15432:db.internal:5432 user@jump-host

Docker / 服务监听

# 只暴露给本机反代：推荐用于单机部署
docker run -p 127.0.0.1:3000:3000 app

# 暴露到所有网卡：公网机器上要谨慎
docker run -p 3000:3000 app

# 看 Linux 上谁在监听端口
ss -ltnp
ss -ltnp 'sport = :3000'

DNS / HTTP / TLS 排障

# DNS：看解析结果和 TTL
dig app.example.com A +short
dig app.example.com CNAME

# HTTP/TLS：看 DNS、连接、证书、请求头、响应头
curl -v https://app.example.com/

# 绕过 DNS，把域名临时指到某个 IP；仍保留 Host/SNI
curl -v --resolve app.example.com:443:203.0.113.10 https://app.example.com/

# 只测试 TCP 连接，不发送 HTTP 请求
curl -v telnet://app.example.com:443

抓包最小集

# 只看某主机和端口的 TCP 包
sudo tcpdump -i any host 203.0.113.10 and tcp port 443

# 抓到文件，稍后用 Wireshark 打开
sudo tcpdump -i any -w app-443.pcap host 203.0.113.10 and tcp port 443
